CREATE TABLE `Workedup`.`product` ( `prodId` INT(4) NOT NULL , `prodName` VARCHAR(30) NOT NULL , `prodPicName` VARCHAR(50) NOT NULL , `prodDescrip` VARCHAR(1000) NULL , `prodPrice` DECIMAL(6,2) NOT NULL DEFAULT '0.00' , `prodQuantity` INT(4) NOT NULL DEFAULT '100' , PRIMARY KEY (`prodId`)) ENGINE = InnoDB
